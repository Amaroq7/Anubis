35  // Specifies priorities for hooks call order in the chain.
36  // For equal priorities first registered hook will be called first.
37  enum class HookPriority : std::uint8_t
38  {
39  UNINTERRUPTABLE = 255, // Hook will be called before other hooks.
40  HIGH = 192, // Hook will be called before hooks with default priority.
41  DEFAULT = 128, // Default hook call priority.
42  MEDIUM = 64, // Hook will be called after hooks with default priority.
43  LOW = 0, // Hook will be called after all other hooks.
44  };
